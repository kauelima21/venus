import { Meta, ColorPalette } from '@storybook/addon-docs';
import { colors, lightColors } from './colors.ts';

<Meta title="Docs/Colors" />

<style>
  {`
    .colors {
      width: 100%;
      padding: 2rem;
    }

    .colors div {
      display: flex;
      justify-content: space-between;
    }

    .colors div strong, span {
      font-size: .875rem;
    }

    .light {
      color: #FFFFFF;
    }
  `}
</style>


<ColorPallete>
  {Object.entries(colors).map(([key, value]) => {
    let className = "colors light";
    if (lightColors.includes(key)) {
      className = "colors";
    }
    return (
      <div key={key} className={className} style={{background: `${value}`}}>
        <div>
          <strong>{key}</strong>
          <span>{value}</span>
        </div>
      </div>
    )
  })}
</ColorPallete>
